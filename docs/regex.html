#!/bin/bash

# ------------------------------------------------------------------------------
# Find a way to select IP addresses like 6.216.166.67, 127.0.0.1, 192.168.0.15
# from text below using RegularExpression.
# ------------------------------------------------------------------------------

touch /tmp/regex.txt

echo \
  "I’m sure there are better subs to post this to but I don’t know them, anyways,\
  my friend keeps getting these weird messages that kinda looks like IP address.\
  This is the entirety of messages, “6.216.166.67:8012/0330024645600028000040000”\
  they send them periodically 192.168.0.1:9200 and the numbers are slightly\
  different but in the same format. I can’t seem to find anything from my\
  phone. Any ideas what this could be?" > /tmp/regex.txt

# -E Interpret PATTERNS as extended regular expressions

grep -E \
  '(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)(\.(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)){3}' \
  /tmp/regex.txt
