// Logic

"ReplaceValue": { // Object with settings for string (value) replacement
  "replaceItems": [ // Array with settings for objects
    {
      "subject": [ // Array with objects that will apply settings
        "Basement" // Object name
      ],
      "search": "N/A", // Search for "N/A"
      "replace": "" // Replace "N/A" with empty string
    },
    {
      "subject": [ // Array with objects that will apply settings
        "View", // Object name
        "Heating", // Object name
        "Basement", // Object name
        "FireplaceFeatures", // Object name
        "ParkingFeatures", // Object name
        "OtherEquipment", // Object name
        "LotFeatures", // Object name
        "Cooling", // Object name
        "AssociationAmenities", // Object name
        "WaterSource", // Object name
        "LaundryFeatures" // Object name
      ],
      "search": "None", // Search for "None"
      "replace": "" // Replace "None" with empty string
    },
    {
      "subject": [ // Array with objects that will apply settings
        "WaterfrontFeatures" // Object name
      ],
      "search": "None,?", // Search for "None,?"
      "regex": true, // There is a flag "regex" that is set to true
                     // we must interpret "search" as regular expression
                     // with values "None" and "None,"
      "replace": "" // Replace "None" and "None," with empty string
    }
  ]
}
